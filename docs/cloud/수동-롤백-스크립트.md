### moyeobab-api 수동 롤백 실행 스크립트

- 용도: CD 자동 롤백 실패 또는 운영 중 긴급 장애 발생 시 수동 복구

- 대상: moyeobab-api.service

#### 백업 파일 확인 → 서비스 중지 → JAR 교체 → 서비스 시작 → 헬스체크 순서로 진행하고, 롤백 실패 시 자동으로 이전 상태 복구를 시도

```
#!/bin/bash

# =============================================================================
# moyeobab-api 수동 롤백 스크립트
# =============================================================================
# 용도: CD 자동 롤백 실패 또는 운영 중 장애 발생 시 수동 롤백
# 대상: moyeobab-api.service
# =============================================================================

set -e

# 설정
SERVICE_NAME="moyeobab-api"
DEPLOY_PATH="/home/ubuntu/app"
JAR_NAME="app.jar"
BACKUP_PATH="/home/ubuntu/app/backup"
BACKUP_JAR="app.jar.backup"
HEALTH_CHECK_URL="https://api.moyeobab.com/actuator/health"
MAX_RETRY=10
RETRY_INTERVAL=3

# 색상 출력
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# 백업 파일 존재 확인
check_backup() {
    log_info "백업 파일 확인 중..."
    if [ ! -f "$BACKUP_PATH/$BACKUP_JAR" ]; then
        log_error "백업 파일이 존재하지 않습니다: $BACKUP_PATH/$BACKUP_JAR"
        exit 1
    fi
    log_info "백업 파일 확인 완료"
}

# plain.jar 여부 검증 (Spring Boot 실행 가능한 JAR인지 확인)
validate_jar() {
    log_info "JAR 파일 유효성 검증 중..."
    
    # 파일명에 plain이 포함되어 있는지 확인
    if echo "$BACKUP_PATH/$BACKUP_JAR" | grep -qi "plain"; then
        log_error "plain.jar는 실행할 수 없습니다. 올바른 bootJar를 사용하세요."
        exit 1
    fi
    
    # unzip 명령어 존재 여부 확인
    if ! command -v unzip &> /dev/null; then
        log_warn "unzip이 설치되어 있지 않아 JAR 내부 검증을 스킵합니다."
        log_warn "sudo apt install unzip 으로 설치하면 검증이 가능합니다."
        return 0
    fi
    
    # Spring Boot 실행 가능한 JAR인지 확인 (BOOT-INF 디렉토리 존재 여부)
    if ! unzip -l "$BACKUP_PATH/$BACKUP_JAR" 2>/dev/null | grep -q "BOOT-INF"; then
        log_error "유효한 Spring Boot JAR가 아닙니다. (BOOT-INF 디렉토리 없음)"
        log_error "plain.jar가 아닌 bootJar로 빌드된 JAR를 사용하세요."
        exit 1
    fi
    
    log_info "JAR 파일 유효성 검증 완료 (Spring Boot 실행 가능)"
}

# 서비스 중지
stop_service() {
    log_info "서비스 중지 중: $SERVICE_NAME"
    sudo systemctl stop $SERVICE_NAME
    
    if systemctl is-active --quiet $SERVICE_NAME; then
        log_error "서비스 중지 실패"
        exit 1
    fi
    log_info "서비스 중지 완료"
}

# JAR 파일 교체
replace_jar() {
    log_info "JAR 파일 교체 중..."
    
    # 현재 JAR 백업 (롤백 실패 대비)
    if [ -f "$DEPLOY_PATH/$JAR_NAME" ]; then
        cp "$DEPLOY_PATH/$JAR_NAME" "$DEPLOY_PATH/$JAR_NAME.rollback_backup"
        log_info "현재 JAR 임시 백업 완료"
    fi
    
    # 백업 JAR로 교체
    cp "$BACKUP_PATH/$BACKUP_JAR" "$DEPLOY_PATH/$JAR_NAME"
    log_info "JAR 파일 교체 완료"
}

# 서비스 시작
start_service() {
    log_info "서비스 시작 중: $SERVICE_NAME"
    sudo systemctl start $SERVICE_NAME
    
    sleep 2
    
    if ! systemctl is-active --quiet $SERVICE_NAME; then
        log_error "서비스 시작 실패"
        log_warn "롤백 이전 상태로 복구를 시도합니다..."
        restore_rollback_backup
        exit 1
    fi
    log_info "서비스 시작 완료"
}

# 헬스체크
health_check() {
    log_info "헬스체크 진행 중..."
    
    for i in $(seq 1 $MAX_RETRY); do
        echo -n "  시도 $i/$MAX_RETRY... "
        
        HTTP_CODE=$(curl -s -o /dev/null -w '%{http_code}' --max-time 10 $HEALTH_CHECK_URL || echo "000")
        
        if [ "$HTTP_CODE" = "200" ]; then
            echo -e "${GREEN}성공${NC} (HTTP: $HTTP_CODE)"
            log_info "헬스체크 통과"
            return 0
        fi
        
        echo -e "${YELLOW}실패${NC} (HTTP: $HTTP_CODE)"
        sleep $RETRY_INTERVAL
    done
    
    log_error "헬스체크 실패 - 서비스가 정상적으로 응답하지 않습니다"
    return 1
}

# 롤백 실패 시 복구
restore_rollback_backup() {
    if [ -f "$DEPLOY_PATH/$JAR_NAME.rollback_backup" ]; then
        log_warn "롤백 이전 JAR로 복구 중..."
        cp "$DEPLOY_PATH/$JAR_NAME.rollback_backup" "$DEPLOY_PATH/$JAR_NAME"
        sudo systemctl start $SERVICE_NAME
        log_warn "복구 완료 - 수동 확인이 필요합니다"
    fi
}

# 임시 백업 정리
cleanup() {
    if [ -f "$DEPLOY_PATH/$JAR_NAME.rollback_backup" ]; then
        rm -f "$DEPLOY_PATH/$JAR_NAME.rollback_backup"
        log_info "임시 백업 파일 정리 완료"
    fi
}

# 서비스 상태 출력
show_status() {
    echo ""
    log_info "===== 서비스 상태 ====="
    systemctl status $SERVICE_NAME --no-pager -l | head -15
    echo ""
}

# 메인 실행
main() {
    echo ""
    echo "============================================="
    echo "  moyeobab-api 수동 롤백 스크립트"
    echo "============================================="
    echo ""
    
    # 확인 프롬프트
    read -p "롤백을 진행하시겠습니까? (y/N): " confirm
    if [ "$confirm" != "y" ] && [ "$confirm" != "Y" ]; then
        log_info "롤백이 취소되었습니다"
        exit 0
    fi
    
    echo ""
    
    # 롤백 실행
    check_backup
    validate_jar
    stop_service
    replace_jar
    start_service
    
    if health_check; then
        cleanup
        show_status
        echo ""
        log_info "============================================="
        log_info "  롤백 완료!"
        log_info "  Discord에 상황을 공유해주세요."
        log_info "============================================="
    else
        show_status
        echo ""
        log_error "============================================="
        log_error "  롤백 후 헬스체크 실패"
        log_error "  서버 상태를 직접 확인해주세요."
        log_error "============================================="
        exit 1
    fi
}

main "$@"
```

### recommend 롤백 스크립트
- 용도: CD 자동 롤백 실패 또는 운영 중 장애 발생 시 수동 복구

- 대상: recommend.service

#### 최근 커밋 확인 → 현재 상태 백업 → 서비스 중지 → Git 롤백 (HEAD~1) → 서비스 시작 → 헬스체크, 롤백 실패 시 자동으로 이전 상태 복구를 시도

#### 파일 위치:`/home/ubuntu/scripts/rollback-recommend.sh`

```bash
#!/bin/bash

# =============================================================================
# recommend 수동 롤백 스크립트
# =============================================================================
# 용도: CD 자동 롤백 실패 또는 운영 중 장애 발생 시 수동 롤백
# 대상: recommend.service
# =============================================================================

set -e

# 설정
SERVICE_NAME="recommend"
REPO_PATH="/home/ubuntu/13-team-project-ai"
SERVICE_DIR="services/recommend"
HEALTH_CHECK_URL="http://127.0.0.1:8000/health"
MAX_RETRY=10
RETRY_INTERVAL=3

# 색상 출력
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

show_commits() {
    log_info "최근 커밋 목록:"
    cd "$REPO_PATH/$SERVICE_DIR"
    git log --oneline -5
    echo ""
}

backup_current_state() {
    log_info "현재 커밋 백업 중..."
    cd "$REPO_PATH/$SERVICE_DIR"
    git rev-parse HEAD > /tmp/recommend_rollback_backup_sha
    log_info "백업 완료: $(cat /tmp/recommend_rollback_backup_sha)"
}

stop_service() {
    log_info "서비스 중지 중: $SERVICE_NAME"
    sudo systemctl stop $SERVICE_NAME
    
    if systemctl is-active --quiet $SERVICE_NAME; then
        log_error "서비스 중지 실패"
        exit 1
    fi
    log_info "서비스 중지 완료"
}

rollback_git() {
    log_info "이전 커밋으로 롤백 중..."
    cd "$REPO_PATH/$SERVICE_DIR"
    
    CURRENT_SHA=$(git rev-parse HEAD)
    git reset --hard HEAD~1
    NEW_SHA=$(git rev-parse HEAD)
    
    log_info "롤백 완료: $CURRENT_SHA → $NEW_SHA"
}

start_service() {
    log_info "서비스 시작 중: $SERVICE_NAME"
    sudo systemctl start $SERVICE_NAME
    
    sleep 2
    
    if ! systemctl is-active --quiet $SERVICE_NAME; then
        log_error "서비스 시작 실패"
        log_warn "롤백 이전 상태로 복구를 시도합니다..."
        restore_rollback_backup
        exit 1
    fi
    log_info "서비스 시작 완료"
}

health_check() {
    log_info "헬스체크 진행 중..."
    
    for i in $(seq 1 $MAX_RETRY); do
        echo -n "  시도 $i/$MAX_RETRY... "
        
        HTTP_CODE=$(curl -s -o /dev/null -w '%{http_code}' --max-time 10 $HEALTH_CHECK_URL || echo "000")
        
        if [ "$HTTP_CODE" = "200" ]; then
            echo -e "${GREEN}성공${NC} (HTTP: $HTTP_CODE)"
            log_info "헬스체크 통과"
            return 0
        fi
        
        echo -e "${YELLOW}실패${NC} (HTTP: $HTTP_CODE)"
        sleep $RETRY_INTERVAL
    done
    
    log_error "헬스체크 실패 - 서비스가 정상적으로 응답하지 않습니다"
    return 1
}

restore_rollback_backup() {
    if [ -f "/tmp/recommend_rollback_backup_sha" ]; then
        log_warn "롤백 이전 상태로 복구 중..."
        cd "$REPO_PATH/$SERVICE_DIR"
        git reset --hard $(cat /tmp/recommend_rollback_backup_sha)
        sudo systemctl start $SERVICE_NAME
        log_warn "복구 완료 - 수동 확인이 필요합니다"
    fi
}

show_status() {
    echo ""
    log_info "===== 서비스 상태 ====="
    systemctl status $SERVICE_NAME --no-pager -l | head -15
    echo ""
}

main() {
    echo ""
    echo "============================================="
    echo "  recommend 수동 롤백 스크립트"
    echo "============================================="
    echo ""
    
    show_commits
    
    read -p "직전 커밋으로 롤백을 진행하시겠습니까? (y/N): " confirm
    if [ "$confirm" != "y" ] && [ "$confirm" != "Y" ]; then
        log_info "롤백이 취소되었습니다"
        exit 0
    fi
    
    echo ""
    
    backup_current_state
    stop_service
    rollback_git
    start_service
    
    if health_check; then
        show_status
        echo ""
        log_info "============================================="
        log_info "  롤백 완료!"
        log_info "  Discord에 상황을 공유해주세요."
        log_info "============================================="
    else
        show_status
        echo ""
        log_error "============================================="
        log_error "  롤백 후 헬스체크 실패"
        log_error "  서버 상태를 직접 확인해주세요."
        log_error "============================================="
        exit 1
    fi
}

main "$@"

### 롤백 실패 시 서버 상태 진단용 실행 스크립트
#### 디스크, 메모리, CPU, 주요 서비스 상태, DB/Redis 연결, 최근 로그를 한 번에 확인가능
```
#!/bin/bash

# =============================================================================
# 서버 상태 확인 스크립트
# =============================================================================
# 용도: 롤백 실패 시 서버 상태 진단
# =============================================================================

# 색상 출력
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC='\033[0m'

section() {
    echo ""
    echo -e "${CYAN}===== $1 =====${NC}"
}

check_status() {
    local name=$1
    local status=$2
    
    if [ "$status" = "active" ]; then
        echo -e "  $name: ${GREEN}$status${NC}"
    else
        echo -e "  $name: ${RED}$status${NC}"
    fi
}

echo ""
echo "============================================="
echo "  서버 상태 진단"
echo "============================================="

# 디스크 확인
section "디스크 사용량"
df -h | grep -E '^/dev|Filesystem'

DISK_USAGE=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')
if [ "$DISK_USAGE" -gt 90 ]; then
    echo -e "${RED}[경고] 디스크 사용량이 90%를 초과했습니다!${NC}"
fi

# 메모리 확인
section "메모리 사용량"
free -h

MEM_USAGE=$(free | grep Mem | awk '{printf "%.0f", $3/$2 * 100}')
if [ "$MEM_USAGE" -gt 90 ]; then
    echo -e "${RED}[경고] 메모리 사용량이 90%를 초과했습니다!${NC}"
fi

# CPU 확인
section "CPU 사용량 (상위 5개 프로세스)"
ps aux --sort=-%cpu | head -6

# 주요 서비스 상태
section "주요 서비스 상태"
services=("moyeobab-api" "recommend" "nginx" "postgresql@16-main" "redis-server")

for svc in "${services[@]}"; do
    status=$(systemctl is-active $svc 2>/dev/null || echo "not found")
    check_status "$svc" "$status"
done

# DB 연결 확인
section "PostgreSQL 연결 확인"
if sudo -u postgres psql -c "SELECT 1;" > /dev/null 2>&1; then
    echo -e "  PostgreSQL: ${GREEN}연결 성공${NC}"
else
    echo -e "  PostgreSQL: ${RED}연결 실패${NC}"
fi

# Redis 연결 확인
section "Redis 연결 확인"
if redis-cli ping > /dev/null 2>&1; then
    echo -e "  Redis: ${GREEN}연결 성공${NC}"
else
    echo -e "  Redis: ${RED}연결 실패${NC}"
fi

# Java 프로세스 확인
section "Java 프로세스"
ps aux | grep java | grep -v grep || echo "  실행 중인 Java 프로세스 없음"

# 최근 로그 확인
section "moyeobab-api 최근 로그 (마지막 20줄)"
sudo journalctl -u moyeobab-api --no-pager -n 20

echo ""
echo "============================================="
echo "  진단 완료"
echo "============================================="
echo ""
